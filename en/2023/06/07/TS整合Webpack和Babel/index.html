Modify Webpack configuration file

loader: Set loader
options: Set predefined environment
targets: Browsers to be compatible with
corejs: Version of corejs used by babel
useBuildIns: How babel calls corejs

usage: Load on demand

environment: Configure Webpack packaging environment

allowFunction: Use arrow functions

false: Do not use arrow functions

const path = require('path');
const HTMLWebpackPlugin = require('html-webpack-plugin')

module.exports = {
  entry: "./src/index.ts",
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: "bundle.js"
  },
  module: {
    rules: [
      {
        test: /\.ts$/,
        use: "ts-loader",
        exclude: /node-modules/
      }
    ]
  },
  plugins: [
    new CleanWebpackPlugin(),
    new HTMLWebpackPlugin({
      template: "./src/index.html"
    }),
    resolve: [
      extensions: ['.ts', '.js']
    ]
  ]
}
